Sample Users API Working!

GET /users

show all users in db

response json

[{"_id":"63a1d297fdef0cbed532bab3","username":"harry","roles":["employee"],"active":true,"__v":1},{"_id":"63a2902a57e9d4369b634d0e","username":"tom","roles":["clerk","admin"],"active":false,"__v":1},{"_id":"63a292c457e9d4369b634d1e","username":"paul","roles":["employee"],"active":true,"__v":0}]

POST /users

create a new user in db
request json { "username": "paul", "password": "p!2022aul", "roles": ["employee"] }
response json {"message":"New user paul created"}

PATCH /users

update existing user in db
request json { "id": "63a292c457e9d4369b634d1e", "username": "paul mayar", "password": "p!2022aul2", "roles": [ "employee" ], "active": false }
response json {"message":"paul mayar updated"}

DELETE /users

delete existing user in db
request json { "id": "63a292c457e9d4369b634d1e" }
response json "Username paul mayar with ID 63a292c457e9d4369b634d1e deleted"
